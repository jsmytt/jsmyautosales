function handleEnter (field, event) {
 var keyCode = event.keyCode ? event.keyCode : event.which ? event.which : event.charCode;
 if (keyCode == 13) {
 var i;
 for (i = 0; i < field.form.elements.length; i++)
 if (field == field.form.elements[i])
 break;
 i = (i + 1) % field.form.elements.length;
 field.form.elements[i].focus();
 return false;
 }
 else
 return true;
 }



function clearFields() {
    var frm = document.getElementById('form');
    var em = frm.elements;
    frm.reset();
    for(var i=0; i<em.length; i++) {
        if(em[i].type == 'text') em[i].value = '';
        if(em[i].type == 'textarea') em[i].value = '';
    }
    return;
}


function calculate(frm){
    var reg, reg1, lease, lease1, drive_lease, sell_cost, cost, acc, acc_sub, acc_sub1, mos_mf, mos_mf1, mos_mf2, tax_adj,tax_adj1,lease_up,lease_down,l_sales_tax, lease_residual;
    var sales_tax_amt, lease_tax, apr, lease_residual1, num;

	if (parseFloat(frm.elements[2].value) < 5000) {
		reg = parseFloat(frm.elements[2].value) * 0.006496 + 110 + 25;
	} else if (parseFloat(frm.elements[2].value) < 25000) {
		reg = parseFloat(frm.elements[2].value) * 0.006496 + 110 + 50;
	} else if (parseFloat(frm.elements[2].value) < 35000) {
		reg = parseFloat(frm.elements[2].value) * 0.006496 + 110 + 100;
	} else if (parseFloat(frm.elements[2].value) < 60000) {
		reg = parseFloat(frm.elements[2].value) * 0.006496 + 110 + 150;
	} else {
		reg = parseFloat(frm.elements[2].value) * 0.006496 + 110 + 175;
	}


	l_sales_tax = 1 + (parseFloat(frm.elements[9].value) / 100);
	tax_adj = 1 - (parseFloat(frm.elements[9].value) / 100);
	tax_adj1 = 1 - (parseFloat(frm.elements[9].value) / 100 * parseFloat(frm.elements[9].value) / 100);
	cost = parseFloat(frm.elements[8].value) + reg + 217.75;
	acc_sub1 = cost - parseFloat(frm.elements[6].value) - parseFloat(frm.elements[7].value);
	sell_cost = parseFloat(frm.elements[2].value) + acc_sub1;
	acc_sub = tax_adj * acc_sub1;
	acc = parseFloat(frm.elements[2].value) + acc_sub;
	mos_mf = parseFloat(frm.elements[5].value) * parseFloat(frm.elements[3].value);
	mos_mf1 = mos_mf + 1;
	mos_mf2 = mos_mf - 1;


	if (parseFloat(frm.elements[4].value) < 100) {
		lease_residual = (parseFloat(frm.elements[1].value)) * (parseFloat(frm.elements[4].value)/100);
		lease_residual1 = parseFloat(frm.elements[4].value)/100 ;

		document.getElementById("result2").innerHTML = "$ " + lease_residual.toFixed(2);

		} else {

		lease_residual = parseFloat(frm.elements[4].value);
		lease_residual1 = (parseFloat(frm.elements[4].value) / parseFloat(frm.elements[1].value)) * 100;

		document.getElementById("result2").innerHTML = lease_residual1.toFixed(2) + "%";
	}


	drive_lease = parseFloat(frm.elements[6].value) + parseFloat(frm.elements[7].value);
	apr = parseFloat(frm.elements[3].value) * 2400;


	if (drive_lease <= cost) {

		lease_up = (sell_cost * mos_mf1) + (lease_residual * mos_mf2);
		lease_down = parseFloat(frm.elements[5].value) - (l_sales_tax * mos_mf1);
		lease = lease_up / lease_down;
		lease_tax = lease * (l_sales_tax - 1);
		lease1 = lease * l_sales_tax;


		} else {

		lease_up = (acc * mos_mf1) + (lease_residual * mos_mf2);
		lease_down = parseFloat(frm.elements[5].value) - (tax_adj1 * mos_mf1);
		lease = lease_up / lease_down;
		lease_tax = lease * (l_sales_tax - 1);
		lease1 = lease * l_sales_tax;

	}

	document.getElementById("result").innerHTML = "$ " + lease.toFixed(2) + " + TAX ($"+ lease_tax.toFixed(2) +")" + "= $ " + lease1.toFixed(2);
    document.getElementById("result0").innerHTML = "$ " + reg.toFixed(2);
   	document.getElementById("result9").innerHTML = apr.toFixed(2) + "%";



}
